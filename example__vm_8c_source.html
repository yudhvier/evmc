77 static struct evmc_result execute(struct evmc_vm* instance,
78  struct evmc_host_context* context,
79  enum evmc_revision rev,
80  const struct evmc_message* msg,
81  const uint8_t* code,
82  size_t code_size)
83 {
84  struct evmc_result ret = {.status_code = EVMC_INTERNAL_ERROR};
85  if (code_size == 0)
86  {
87  // In case of empty code return a fancy error message.
88  const char* error = rev == EVMC_BYZANTIUM ? "Welcome to Byzantium!" : "Hello Ethereum!";
89  ret.output_data = (const uint8_t*)error;
90  ret.output_size = strlen(error);
91  ret.status_code = EVMC_FAILURE;
92  ret.gas_left = msg->gas / 10;
93  ret.release = NULL; // We don't need to release the constant messages.
94  return ret;
95  }

97  struct example_vm* vm = (struct example_vm*)instance;
98 
99  // Simulate executing by checking for some code patterns.
100  // Solidity inline assembly is used in the examples instead of EVM bytecode.
101 
102  // Assembly: `{ mstore(0, address()) return(0, msize()) }`.
103  const char return_address[] = "\x30\x60\x00\x52\x59\x60\x00\xf3";
104 
105  // Assembly: `{ sstore(0, add(sload(0), 1)) }`
106  const char counter[] = "\x60\x01\x60\x00\x54\x01\x60\x00\x55";
107 
108  // Assembly: `{ mstore(0, number()) return(0, msize()) }`
109  const char return_block_number[] = "\x43\x60\x00\x52\x59\x60\x00\xf3";
110 
111  // Assembly: `{ sstore(0, number()) mstore(0, number()) return(0, msize()) }`
112  const char save_return_block_number[] = "\x43\x60\x00\x55\x43\x60\x00\x52\x59\x60\x00\xf3";
113 
114  // Assembly: PUSH(0) 6x DUP1 CALL
115  const char make_a_call[] = "\x60\x00\x80\x80\x80\x80\x80\x80\xf1";
116 
117  if (msg->kind == EVMC_CREATE)
118  {
119  ret.status_code = EVMC_SUCCESS;
120  ret.gas_left = msg->gas / 10;
121  return ret;
122  }
123  else if (code_size == (sizeof(return_address) - 1) &&
124  strncmp((const char*)code, return_address, code_size) == 0)
125  {
126  static const size_t address_size = sizeof(msg->destination);
127  uint8_t* output_data = (uint8_t*)malloc(address_size);
128  if (!output_data)
129  {
130  // malloc failed, report internal error.
131  ret.status_code = EVMC_INTERNAL_ERROR;
132  return ret;
133  }
134  memcpy(output_data, &msg->destination, address_size);
135  ret.status_code = EVMC_SUCCESS;
136  ret.output_data = output_data;
137  ret.output_size = address_size;
138  ret.release = &free_result_output_data;
139  return ret;
140  }
141  else if (code_size == (sizeof(counter) - 1) &&
142  strncmp((const char*)code, counter, code_size) == 0)
143  {
144  const evmc_bytes32 key = {{0}};
145  evmc_bytes32 value = context->host->get_storage(context, &msg->destination, &key);
146  value.bytes[31]++;
147  context->host->set_storage(context, &msg->destination, &key, &value);
148  ret.status_code = EVMC_SUCCESS;
149  return ret;
150  }
151  else if (code_size == (sizeof(return_block_number) - 1) &&
152  strncmp((const char*)code, return_block_number, code_size) == 0)
153  {
154  const struct evmc_tx_context tx_context = context->host->get_tx_context(context);
155  const size_t output_size = 20;
156 
157  uint8_t* output_data = (uint8_t*)calloc(1, output_size);
158  snprintf((char*)output_data, output_size, "%u", (unsigned)tx_context.block_number);
159  ret.status_code = EVMC_SUCCESS;
160  ret.gas_left = msg->gas / 2;
161  ret.output_data = output_data;
162  ret.output_size = output_size;
163  ret.release = &free_result_output_data;
164  return ret;
165  }
166  else if (code_size == (sizeof(save_return_block_number) - 1) &&
167  strncmp((const char*)code, save_return_block_number, code_size) == 0)
168  {
169  const struct evmc_tx_context tx_context = context->host->get_tx_context(context);
170  const size_t output_size = 20;
171 
172  // Store block number.
173  const evmc_bytes32 key = {{0}};
174  evmc_bytes32 value = {{0}};
175  // NOTE: assume block number is <= 255
176  value.bytes[31] = (uint8_t)tx_context.block_number;
177  context->host->set_storage(context, &msg->destination, &key, &value);
178 
179  // Return block number.
180  uint8_t* output_data = (uint8_t*)calloc(1, output_size);
181  snprintf((char*)output_data, output_size, "%u", (unsigned)tx_context.block_number);
182  ret.status_code = EVMC_SUCCESS;
183  ret.gas_left = msg->gas / 2;
184  ret.output_data = output_data;
185  ret.output_size = output_size;
186  ret.release = &free_result_output_data;
187  return ret;
188  }
189  else if (code_size == (sizeof(make_a_call) - 1) &&
190  strncmp((const char*)code, make_a_call, code_size) == 0)
191  {
192  struct evmc_message call_msg;
193  memset(&call_msg, 0, sizeof(call_msg));
194  call_msg.kind = EVMC_CALL;
195  call_msg.depth = msg->depth + 1;
196  call_msg.gas = msg->gas - (msg->gas / 64);
197  call_msg.sender = msg->destination;
198  return context->host->call(context, &call_msg);
199  }
200 
201  ret.status_code = EVMC_FAILURE;
202  ret.gas_left = 0;
203 
204  if (vm->verbose)
205  printf("Execution done.\n");
206 
207  return ret;
208 }
