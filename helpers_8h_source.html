120 static inline struct evmc_result evmc_make_result(enum evmc_status_code status_code,
121  int64_t gas_left,
122  const uint8_t* output_data,
123  size_t output_size)
124 {
125  struct evmc_result result;
126  memset(&result, 0, sizeof(result));
127 
128  if (output_size != 0)
129  {
130  uint8_t* buffer = (uint8_t*)malloc(output_size);
131 
132  if (!buffer)
133  {
134  result.status_code = EVMC_OUT_OF_MEMORY;
135  return result;
136  }
137 
138  memcpy(buffer, output_data, output_size);
139  result.output_data = buffer;
140  result.output_size = output_size;
141  result.release = evmc_free_result_memory;
142  }
143 
144  result.status_code = status_code;
145  result.gas_left = gas_left;
146  return result;
147 }
